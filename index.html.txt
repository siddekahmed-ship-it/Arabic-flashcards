  <div id="flashcard">
    <div class="card" id="card">
      <div class="card-content card-front" id="card-front"></div>

    </div>

  <div class="controls">
    <button onclick="prevCard()">Previous</button>
    <button onclick="flipCard()">Flip</button>
    <button onclick="nextCard()">Next</button>
    <button onclick="randomCard()">Random</button>
  </div>
  <script>
    // Sample data (replace with your full list for production)
    const cards = [
      {
        Arabic: "أَهْلٌ",
        Root: "أَهْلَ",
        Grammatical: "Noun (singular)",
        English: "Family"
      },
      {
        Arabic: "آبَاءُ",
        Root: "أَبَ",
        Grammatical: "Noun (plural)",
        English: "Fathers, ancestors"
      },
      {
        Arabic: "جَنَّةٌ",
        Root: "جَنَنَ",
        Grammatical: "Noun (singular)",
        English: "Garden, Paradise"
      },
      {
        Arabic: "حَيَاةٌ",
        Root: "حَيَيَ",
        Grammatical: "Noun (fem. singular)",
        English: "Life"
      },
      {
        Arabic: "إِيمَانٌ",
        Root: "أَمَنَ",
        Grammatical: "Noun (verbal noun)",
        English: "Faith, belief"
      }
      // ... Add more entries here or load from a file
    ];

    let current = 0;
    let flipped = false;

    function showCard(idx) {
      const card = cards[idx];
      document.getElementById('card-front').innerHTML = `<span style="font-size:2.5em; color:#16a085;">${card.Arabic}</span>`;
      document.getElementById('card-back').innerHTML = `
        <b>${card.English}</b>
        <div style="margin-top:10px; font-size:1em;">
          <i>${card.Grammatical}</i><br>
          <span style="color:#888;">Root: ${card.Root}</span>
        </div>
      `;
      document.getElementById('progress').textContent = `Card ${idx + 1} of ${cards.length}`;
      document.getElementById('card').classList.remove('flipped');
      flipped = false;
    }

    function flipCard() {
      const cardDiv = document.getElementById('card');
      cardDiv.classList.toggle('flipped');
      flipped = !flipped;
    }

    function nextCard() {
      current = (current + 1) % cards.length;
      showCard(current);
    }

    function prevCard() {
      current = (current - 1 + cards.length) % cards.length;
      showCard(current);
    }

    function randomCard() {
      let idx;
      do {
        idx = Math.floor(Math.random() * cards.length);
      } while (idx === current && cards.length > 1);
      current = idx;
      showCard(current);
    }

    // Flip on click
    document.getElementById('card').onclick = flipCard;

    // Keyboard navigation
    document.addEventListener('keydown', e => {
      if (e.key === 'ArrowRight') nextCard();
      else if (e.key === 'ArrowLeft') prevCard();
      else if (e.key === ' ') flipCard();
      else if (e.key === 'r') randomCard();
    });

    // Initialize
    showCard(current);
  </script>
